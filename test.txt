3.FINDING LCM OF A LIST:
	arr = list(map(int,input().split()))
	xxx=[]
	for i in range(1,1000):
    	    mx = max(arr)
    	    multiple = mx * i
    	    xxx.append(multiple)
	yyy=[]
	for j in arr:

                if k%j==0:
        	    yyy.append(k)
	import collections
	w=collections.Counter(yyy)
	for k,v in w.items():

5.MAXIMUM OCCURED WORDS IN THE PARAGRAPH:
	para= list(map(str,input().split()))
	import collections
	w=collections.Counter(para)

	mx=[]
	i=1
	for k,v in w.items():
    	    if v>i:
        	i=v
        	mx.append(i)
	b=max(mx)
	ans=[]
	for k,v in w.items():
    	if v==b:
            ans.append(k)
	print('maximum occured words in the paragraphs:',b,'times:','--->',ans)
